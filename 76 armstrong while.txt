#include <stdio.h>
int main()
{
    //input 153 ----> 1*1*1 + 5*5*5 + 3*3*3 = 153
    int n,c,r,arm=0;
    printf("Enter a number: ");
    scanf("%d",&n);
    c=n;
    while(n>0)
    {
    r=n%10;
    arm=arm+(r*r*r);
    n=n/=10;
    }
    {
        if(c==arm)
        printf("%d is a armstrong number",c);
        else
        printf("%d is not a armstrong number",c);
    }
    return 0;
}
