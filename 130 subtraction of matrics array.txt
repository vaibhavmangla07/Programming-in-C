#include <stdio.h>

int main() 
{
    int i, j, a[3][3], b[3][3], c[3][3];

    printf("Enter Elements of matrix A: ");
    for (i = 0; i < 3; i++) 
    {
        for (j = 0; j < 3; j++) 
        {
            scanf("%d", &a[i][j]);
        }
    }

    printf("\nEnter Elements of matrix B: ");
    for (i = 0; i < 3; i++) 
    {
        for (j = 0; j < 3; j++) 
        {
            scanf("%d", &b[i][j]);
        }
    }

    printf("\nSubtraction of matrices:\n");
    for (i = 0; i < 3; i++) 
    {
        for (j = 0; j < 3; j++) 
        {
            c[i][j] = a[i][j] - b[i][j];
        }
    }

    for (i = 0; i < 3; i++) 
    {
        for (j = 0; j < 3; j++) 
        {
            printf("%d ", c[i][j]);
        }
        printf("\n");
    }

    return 0;
}
